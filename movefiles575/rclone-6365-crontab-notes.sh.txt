logrotate to handle old logs.


sudo tee /etc/logrotate.d/albe_rotate01 <<EOF2
# rclone
/ap/log/rclone/*.log  {
     su albe albe
     size=1
     rotate 90
     daily

     #dateext
     #dateformat _%Y-%m-%d
